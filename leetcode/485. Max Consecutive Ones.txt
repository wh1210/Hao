class Solution {
    public int findMaxConsecutiveOnes(int[] nums) {
        int ones = 0;
        int res = 0;
        for (int num : nums) {
            if (num == 1) {
                ones++;
                res = Math.max(ones, res);
            } else {
                ones = 0;
            }
        }
        return res;
    }
}
