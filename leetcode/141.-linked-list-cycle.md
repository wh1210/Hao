# 141. Linked List Cycle

//--------------------------------------------------------------------- /\*\*

* Definition for singly-linked list.
* class ListNode {
* int val;
* ListNode next;
* ListNode\(int x\) {
* val = x;
* next = null;
* }
* }

  \*/

// HashTable public class Solution { public boolean hasCycle\(ListNode head\) { HashSet set = new HashSet&lt;&gt;\(\);

```text
    ListNode curr = head;

    while (curr != null) {
        if (!set.contains(curr)) {
            set.add(curr);
        } else {
            return true;
        }
        curr = curr.next;
    }

    return false;
}
```

}

//----------------------------------------------------------------------------------- // This approach uses no extra space. /\*\*

* Definition for singly-linked list.
* class ListNode {
* int val;
* ListNode next;
* ListNode\(int x\) {
* val = x;
* next = null;
* }
* }

  \*/

// Two-pointers. public class Solution { public boolean hasCycle\(ListNode head\) { if \(head == null\) { return false; }

```text
    ListNode slow = head;
    ListNode fast = head.next;

    while (fast != null && fast.next != null) {
        if (slow == fast) {
            return true;
        }
        slow = slow.next;
        fast = fast.next.next;
    }

    return false;
}
```

}
