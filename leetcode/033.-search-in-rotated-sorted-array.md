# 033. Search in Rotated Sorted Array

class Solution { public int search\(int\[\] nums, int target\) { int lo = 0; int hi = nums.length - 1; while \(lo &lt; hi\) { int mid = \(lo + hi\) / 2; if \(nums\[mid\] &lt; nums\[hi\]\) { hi = mid; } else { lo = mid + 1; } }

```text
    int mod = lo;
    lo = 0; 
    hi = nums.length - 1;
    while (lo <= hi) {
        int mid = (lo + hi) / 2;
        int realMid = (mid + mod) % nums.length;
        if (target > nums[realMid]) {
            lo = mid + 1;
        } else if (target < nums[realMid]){
            hi = mid - 1;
        } else {
            return realMid;
        }
    }

    return -1;

}
```
