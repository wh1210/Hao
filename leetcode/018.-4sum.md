# 018. 4Sum

class Solution { public List&gt; fourSum\(int\[\] nums, int target\) { List&gt; res = new ArrayList&lt;&gt;\(\); Arrays.sort\(nums\); return res = kSum\(nums, 0, 4, target\); }

```text
private List<List<Integer>> kSum(int[] nums, int startIndex, int k, int target) {
    List<List<Integer>> res = new ArrayList<>();
    if (startIndex >= nums.length) return res;
    if (k == 2) {
        int i = startIndex, j = nums.length - 1;

        while (i < j) {
            if (nums[i] + nums[j] == target) {
                ArrayList<Integer> temp = new ArrayList<>();
                temp.add(nums[i]);
                temp.add(nums[j]);
                res.add(temp);

                while (i < j && nums[i] == nums[i + 1]) i++;
                while (i < j && nums[j] == nums[j - 1]) j--;

                i++;
                j--;
            } else  if (nums[i] + nums[j] < target) {
                i++;
            } else {
                j--;
            }
        }
    } else {
        List<List<Integer>> temp = new ArrayList<>();
        for (int i = startIndex; i < nums.length - k + 1; i++) {
            temp = kSum(nums, i + 1, k - 1, target - nums[i]);
            if (temp != null) {
                for (List<Integer> t : temp) {
                    t.add(0, nums[i]);
                }
                res.addAll(temp);
            }
            while (i < nums.length - 1 && nums[i] == nums[i + 1]) i++;
        }
    }

    return res;
}
```
