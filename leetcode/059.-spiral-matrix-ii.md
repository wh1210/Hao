# 059. Spiral Matrix II

```java
class Solution {
    public int[][] generateMatrix(int n) {
        int[][] matrix = new int[n][n];
        int[] dx = new int[] {0, 1, 0, -1};
        int[] dy = new int[] {1, 0, -1, 0};
        int x = 0, y = 0, dir = 0;
        boolean[][] seen = new boolean[n][n];
        for (int i = 1; i <= n * n; i++) {
            matrix[x][y] = i;
            seen[x][y] = true;
            int xx = x + dx[dir];
            int yy = y + dy[dir];
            if (xx < 0 || xx >= n|| yy < 0 || yy >= n || seen[xx][yy]) {
                dir = (dir + 1) % 4;
                x = x + dx[dir];
                y = y + dy[dir];
            } else {
                x = xx;
                y = yy;
            }
        }
        return matrix;
    }
}
```
