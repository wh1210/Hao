# 454. 4Sum II

```java
class Solution {
    public int fourSumCount(int[] A, int[] B, int[] C, int[] D) {
        HashMap<Integer, Integer> map = new HashMap<>();
        int length = A.length;
        for (int i = 0; i < length; i++) {
            for (int j = 0; j < length; j++) {
                int temp = A[i] + B[j];
                map.put(temp, map.getOrDefault(temp, 0) + 1);
            }
        }
        
        int res = 0;
        for (int i = 0; i < length; i++) {
            for (int j = 0; j < length; j++) {
                int temp = C[i] + D[j];
                res += map.getOrDefault(-temp, 0);
            }
        }
        
        return res;
    }
}
```
