//------------------------------------------------------------------
// Two-pointer
// Reverse array in-place
class Solution {
    public void reverseWords(char[] str) {
        reverse(str, 0, str.length - 1);
        int lo = 0;
        int hi = 0;
        for (int i = 0; i < str.length; i++) {
            
            if (str[i] != ' ') {
                hi++;
            } else {
                reverse(str, lo, hi - 1);
                lo = i + 1;
                hi = i + 1;
            }
        }
        reverse(str, lo, hi - 1);
    }
    
    private void reverse(char[] str, int lo, int hi) {
        while (lo < hi) {
            char temp = str[lo];
            str[lo] = str[hi];
            str[hi] = temp;
            lo++;
            hi--;
        }
    }
}
