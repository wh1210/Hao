# 050. Pow\(x, n\)

//----------------------------------------------------------------------------- // T: O\(log\(n\)\) // S: O\(log\(n\)\) class Solution { public double myPow\(double x, int n\) { int N = n; if \(n &lt; 0\) { N = -n; x = 1 / x; } return helper\(x, N\); }

```text
private double helper(double x, int n) {
    if (n == 0) return 1;
    double temp = helper(x, n / 2);
    if (n % 2 == 0) {
        return temp * temp;
    } else {
        return temp * temp * x;
    }
}
```
