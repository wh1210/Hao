class Solution {
    public int longestPalindrome(String s) {
        int[] res = new int[256];
        
        for (int i = 0; i < s.length(); i++) {
            res[s.charAt(i)]++;
        }
        
        int length = 0;
        int longestOdd = 0;
        int hasOdd = 0;
        for (int temp : res) {
            if (temp % 2 != 0) {
                length += (temp - 1);
                hasOdd = 1;
            } else {
                length += temp;
            } 
        }
        
        return length + hasOdd;
    }
}
